LED pertama akan menyala jika nilai hasil baca antara 0 sampai dengan 200,LED kedua menyala untuk hasil baca antara 201-400,
LED ke-3 untuk range nilai antara 401-600, LED ke-4 untuk range nilai 601-800,dan LED ke 5 untuk hasil baca yang lebih dari 800.

Kode Program

int ledPin1 = 13;
int ledPin2 = 12;
int ledPin3 = 11;
int ledPin4 = 10;
int ledPin5 = 9;
int ledPin7 = 7;
int sensorPin = A0;
int sensorValue = 0;
int redupTerang = 0;
void setup() {
// put your setup code here, to run once:
pinMode(ledPin1,OUTPUT);
pinMode(ledPin2,OUTPUT);
pinMode(ledPin3,OUTPUT);
pinMode(ledPin4,OUTPUT);
pinMode(ledPin5,OUTPUT);
}
void loop() {
// put your main code here, to run repeatedly:
sensorValue = analogRead(sensorPin);
if(sensorValue > 0 && sensorValue <= 200){
digitalWrite(ledPin1,HIGH);
digitalWrite(ledPin2,LOW);
analogWrite(ledPin7,
}
else if (sensorValue > 201 && sensorValue <= 400){
digitalWrite(ledPin2,HIGH);
digitalWrite(ledPin1,LOW);
digitalWrite(ledPin3,LOW);
}
else if (sensorValue > 401 && sensorValue <= 600){
digitalWrite(ledPin3,HIGH);
digitalWrite(ledPin2,LOW);
digitalWrite(ledPin4,LOW);
}
else if (sensorValue > 601 && sensorValue < 800){
digitalWrite(ledPin4,HIGH);
digitalWrite(ledPin3,LOW);
digitalWrite(ledPin5,LOW);
}
else if (sensorValue > 800){
digitalWrite(ledPin5,HIGH);
digitalWrite(ledPin4,LOW);
}
}
