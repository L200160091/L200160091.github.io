Menggunakan 7-segment sebagai indikator intensitas cahaya sederhana, dengan LDR sebagai sensornya.

Kode Program

int potPin = A0;
int LEDPin = 9;
int potValue = 0;
int LEDValue = 0;
int val[] = {B11000000,B11111001,B10100100,B10110000,B10011001,B10010010,B10000010,B11111000,B10000000,B10010000};
void setup() {
// put your setup code here, to run once:
DDRD = B11111111 ;
}
void loop() {
// put your main code here, to run repeatedly:
potValue = analogRead(potPin);
LEDValue = potValue;
if(LEDValue > 0 && LEDValue < 100){
PORTD = val[0];
}
else if(LEDValue > 101 && LEDValue < 200){
PORTD = val[1];
}
else if(LEDValue > 201 && LEDValue < 300){
PORTD = val[2];
}
else if(LEDValue > 301 && LEDValue < 400){
PORTD = val[3];
}
else if(LEDValue > 401 && LEDValue < 500){
PORTD = val[4];
}
else if(LEDValue > 501 && LEDValue < 600){
PORTD = val[5];
}
else if(LEDValue > 601 && LEDValue < 700){
PORTD = val[6];
}
else if(LEDValue > 701 && LEDValue < 800){
PORTD = val[7];
}
else if(LEDValue > 801 && LEDValue < 900){
PORTD = val[8];
}
else if(LEDValue > 901 && LEDValue < 1000){
PORTD = val[9];
}
}
